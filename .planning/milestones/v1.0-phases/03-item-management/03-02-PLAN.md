3. In `_ready()`, update the slot instantiation loop to connect `slot_clicked`:
```gdscript
func _ready() -> void:
    for child in grid.get_children():
        child.queue_free()
    for i in range(SLOT_COUNT):
        var slot := SLOT_SCENE.instantiate()
        grid.add_child(slot)
        slot.slot_clicked.connect(_on_slot_clicked.bind(i))
    visible = false
```

4. Add `_on_slot_clicked(index: int)` handler:
```gdscript
func _on_slot_clicked(index: int) -> void:
    if _inventory == null:
        return
    if index >= _inventory.slots.size() or _inventory.slots[index].is_empty():
        return  # no-op for empty slots
    if _selected_index == index:
        _deselect()
    else:
        _deselect()
        _selected_index = index
        grid.get_child(index).set_selected(true)
```

Note: `_on_slot_clicked` receives the slot index because `.bind(i)` is used in _ready(). The signal emits the slot node, but bind prepends the index before the signal args — use `func _on_slot_clicked(index: int, _slot_node) -> void` to accept both, or connect with `slot.slot_clicked.connect(func(_n): _on_slot_clicked(i))`. Use the lambda form to avoid signal-arg count mismatch: `slot.slot_clicked.connect(func(_n): _on_slot_clicked(i))`.

5. Add `_deselect()`:
```gdscript
func _deselect() -> void:
    if _selected_index >= 0 and _selected_index < grid.get_child_count():
        grid.get_child(_selected_index).set_selected(false)
    _selected_index = -1
```

6. Update `_input()` to clear selection on close and handle E key for use:
```gdscript
func _input(event: InputEvent) -> void:
    if event.is_action_pressed("inventory"):
        visible = !visible
        if not visible:
            _deselect()
        return
    if not visible:
        return
    if event.is_action_pressed("interact"):
        _use_selected()
        get_viewport().set_input_as_handled()
```

7. Add `_use_selected()`:
```gdscript
func _use_selected() -> void:
    if _selected_index < 0 or _inventory == null:
        return
    var slot := _inventory.slots[_selected_index]
    if slot.is_empty() or not slot.item.consumable:
        return  # silent no-op for non-consumables
    if not slot.item is HealthItem:
        return
    var heal := (slot.item as HealthItem).health
    _inventory.remove(slot.item, 1)
    if _player != null:
        _player.increase_health(heal)
    if slot.is_empty():
        _deselect()
```

Note on remove-before-heal order: Remove first (so if HP is already max, item is still consumed — matches locked decision that E on consumable always uses). Heal after.

Note: `HealthItem` class is autoloaded via class_name — no explicit preload needed.
